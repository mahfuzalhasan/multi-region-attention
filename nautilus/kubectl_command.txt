
-- create: 
    kubectl apply -f .\train_pod.yaml
    (similar for creating persistent volume / pod / job)

-- get into bash on a running pod/job:
    kubectl exec -it job-name -- /bin/bash

-- get:
    kubectl get pods
    kubectl get pvc    

-- copy files from persistent volume to local:
    kubectl cp <namespace>/<pod-name>:/<path-to-file> <path-to-local>

-- delete: 
    kubectl delete job my-job
     kubectl delete pods --all --force

-- log 
    kubectl logs seg-job

-- event 
    kubectl get events --sort-by='.metadata.creationTimestamp'
    kubectl get events --watch --sort-by='.metadata.creationTimestamp'

-- port forwarding
    kubectl port-forward test-pod 8888:8888
    kubectl port-forward test-pod 8889:8889

-- jupyter 
    jupyter notebook --ip='0.0.0.0' --port=8888 --allow-root

-- check command history 
    kubectl exec <pod-name> -- ps aux
